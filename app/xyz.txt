import requests

from requests.auth import HTTPBasicAuth



from app.history_service import save_virtual_change, save_actual_override

from app.ditto_reader import LEGACY_THING_ID, get_virtual, get_actual, thing_exists



DITTO_BASE_URL = "http://localhost:8080/api/2/things"

AUTH = HTTPBasicAuth("ditto", "ditto")

MERGE_HEADERS = {"Content-Type": "application/merge-patch+json"}





def create_thing(thing_id: str, name: str, field: str = None, crop: str = None):



    payload = {

        "attributes": {

            "type": "farm",

            "name": name,

            "field": field,

            "crop": crop

        },

        "features": {

            "actual": {"properties": {}},

            "virtual": {"properties": {}}

        }

    }



    response = requests.put(

        f"{DITTO_BASE_URL}/{thing_id}",

        auth=AUTH,

        json=payload

    )



    response.raise_for_status()



    return response.json()





def ensure_thing(thing_id: str, name: str = None, field: str = None, crop: str = None):

    """

    Auto-provisions the Ditto thing the first time a farm_id shows up,

    since the dashboard never calls an explicit create-farm route.

    """

    if not thing_exists(thing_id):

        create_thing(thing_id, name or thing_id, field, crop)





def update_virtual_properties(properties: dict, thing_id: str = LEGACY_THING_ID):



    ensure_thing(thing_id)



    current = get_virtual(thing_id)



    response = requests.patch(

        f"{DITTO_BASE_URL}/{thing_id}/features/virtual/properties",

        auth=AUTH,

        json=properties,

        headers=MERGE_HEADERS

    )



    response.raise_for_status()



    changes = {}



    for name, value in properties.items():

        old_value = current.get(name)

        save_virtual_change(name, old_value, value, thing_id=thing_id)

        changes[name] = {"oldValue": old_value, "value": value}



    return {

        "status": "success",

        "changes": changes

    }





def update_actual_properties(properties: dict, thing_id: str = LEGACY_THING_ID):



    ensure_thing(thing_id)



    current = get_actual(thing_id)



    response = requests.patch(

        f"{DITTO_BASE_URL}/{thing_id}/features/actual/properties",

        auth=AUTH,

        json=properties,

        headers=MERGE_HEADERS

    )



    response.raise_for_status()



    changes = {}



    for name, value in properties.items():

        old_value = current.get(name)

        save_actual_override(name, old_value, value, thing_id=thing_id)

        changes[name] = {"oldValue": old_value, "value": value}



    return {

        "status": "success",

        "changes": changes

    }

import requests

from requests.auth import HTTPBasicAuth



DITTO_BASE_URL = "http://localhost:8080/api/2/things"

LEGACY_THING_ID = "smartfarm:twin01"



AUTH = HTTPBasicAuth("ditto", "ditto")





def thing_id_for_farm(farm_id: str) -> str:

    if not farm_id or farm_id == "legacy":

        return LEGACY_THING_ID

    return f"smartfarm:{farm_id}"





def thing_exists(thing_id: str) -> bool:

    response = requests.get(f"{DITTO_BASE_URL}/{thing_id}", auth=AUTH)

    return response.status_code == 200





def get_twin(thing_id: str = LEGACY_THING_ID):

    response = requests.get(f"{DITTO_BASE_URL}/{thing_id}", auth=AUTH)

    response.raise_for_status()

    return response.json()





def get_actual(thing_id: str = LEGACY_THING_ID):

    return get_twin(thing_id).get("features", {}).get("actual", {}).get("properties", {})





def get_virtual(thing_id: str = LEGACY_THING_ID):

    return get_twin(thing_id).get("features", {}).get("virtual", {}).get("properties", {})





def get_attributes(thing_id: str = LEGACY_THING_ID):

    return get_twin(thing_id).get("attributes", {})

```python

import re

import requests

from requests.auth import HTTPBasicAuth



from app.ditto_writer import create_thing



DITTO_SEARCH_URL = "http://localhost:8080/api/2/search/things"

AUTH = HTTPBasicAuth("ditto", "ditto")



FARM_ID_PATTERN = re.compile(r"^smartfarm:farm(\d+)$")





def list_farm_thing_ids():

    response = requests.get(

        DITTO_SEARCH_URL,

        auth=AUTH,

        params={"filter": 'eq(attributes/type,"farm")'},

    )

    response.raise_for_status()

    return [item["thingId"] for item in response.json().get("items", [])]





def next_farm_id() -> str:

    used = []

    for thing_id in list_farm_thing_ids():

        match = FARM_ID_PATTERN.match(thing_id)

        if match:

            used.append(int(match.group(1)))

    next_number = (max(used) + 1) if used else 1

    return f"farm{next_number:02d}"





def create_farm(name: str, field: str = None, crop: str = None):

    farm_id = next_farm_id()

    thing_id = f"smartfarm:{farm_id}"

    create_thing(thing_id, name, field, crop)

    return {"farm_id": farm_id, "thing_id": thing_id, "name": name}
```

```python
import time



from app.ditto_reader import get_actual

from app.farm_registry import list_farm_thing_ids

from app.history_service import save_actual_sensor_history





def start_history_logger():



    while True:



        try:

            thing_ids = list_farm_thing_ids()

        except Exception as e:

            print("could not list farms:", e)

            thing_ids = []



        for thing_id in thing_ids:



            try:

                actual = get_actual(thing_id)

                save_actual_sensor_history(actual, thing_id=thing_id)

                print(f"history saved: {thing_id}")



            except Exception as e:

                print(f"history logger error ({thing_id}):", e)



        time.sleep(30)
```

function timeAgo(ts){

  if(!ts) return 'never';

  const s = Math.floor((Date.now()-ts)/1000);

  if(s<5) return 'just now';

  if(s<60) return s+'s ago';

  const m = Math.floor(s/60);

  if(m<60) return m+'m ago';

  return Math.floor(m/60)+'h ago';

}

function defaultScenarioOverrides(sensors){

  const find = k => sensors.find(s=>s.key===k);

  const mk = (keys, pairs)=>{

    const o={};

    keys.forEach((k,i)=>{ if(find(k)) o[k]={min:pairs[i][0],max:pairs[i][1]}; });

    return o;

  };

  return {

    'Flood': mk(['rainfall','moisture','humidity','floodRisk','droughtRisk'], [[120,200],[45,60],[85,100],[60,100],[0,5]]),

    'Drought': mk(['rainfall','moisture','humidity','airTemperature','droughtRisk','floodRisk'], [[0,2],[5,12],[10,30],[35,46],[60,100],[0,5]]),

    'Heat Stroke': mk(['airTemperature','humidity','moisture','uvIndex'], [[40,46],[10,25],[5,15],[9,12]]),

    'Excessive Cold': mk(['airTemperature','temperature','humidity'], [[-8,2],[0,6],[40,70]]),

  };

}



function createFarm(name, legacy=false, opts={}){

  const sensors = opts.sensors ? JSON.parse(JSON.stringify(opts.sensors)) : defaultSensors();

  const meta = opts.meta ? {...opts.meta} : defaultMeta();

  const defaults = defaultScenarioOverrides(sensors);

  const scenarios = {};

  BASE_SCENARIOS.forEach(s=>{ scenarios[s] = {overrides: defaults[s] || {}, saved:false}; });

  return {

    id: legacy ? 'legacy' : (opts.id || uid('farm')),

    name, legacy,

    field: meta.field, crop: meta.crop,

    sensors, meta, scenarios,

    activeScenario: null,

    actualStale: false,

    lastActualOk: 0,

    timers: {actual:null, virtual:null},

    image: null, // {dataUrl, ts}

    history: [], // [{ts, scenario, snapshot:{key:value}}]

  };

}

/* Combined sensor library pooled from every farm that currently exists.

   Grows as new sensor types are added on any farm. Excludes sensors the

   target farm already has, and is split into actual / virtual groups. */

function sensorLibrary(targetFarm){

  const seen = new Map(); // key -> sensor definition (first one found)

  state.farms.forEach(f=>{

    f.sensors.forEach(s=>{

      if(!seen.has(s.key)) seen.set(s.key, s);

    });

  });

  const have = new Set(targetFarm.sensors.map(s=>s.key));

  const actual = [];

  const virtual = [];

  seen.forEach((s,key)=>{

    if(have.has(key)) return;

    (s.mode==='actual' ? actual : virtual).push(s);

  });

  actual.sort((a,b)=>a.name.localeCompare(b.name));

  virtual.sort((a,b)=>a.name.localeCompare(b.name));

  return {actual, virtual};

}



function addSensorFromLibrary(key){

  const farm = activeFarm(); if(!farm) return;

  if(farm.sensors.some(s=>s.key===key)) return;

  let src = null;

  for(const f of state.farms){

    const found = f.sensors.find(s=>s.key===key);

    if(found){ src = found; break; }

  }

  if(!src) return;

  farm.sensors.push({...src});

  renderContent();

}



/* ---------------- backend I/O (graceful fallback) ---------------- */



function actualUrl(farm){ return farm.legacy ? '/farm/actual' : `/farm/${farm.id}/actual`; }

function virtualUrl(farm){ return farm.legacy ? '/farm/virtual' : `/farm/${farm.id}/virtual`; }



async function tickActual(farm){

  try{

    const res = await fetch(actualUrl(farm));

    if(!res.ok) throw new Error('status '+res.status);

    const data = await res.json();

    farm.sensors.forEach(s=>{

      if(s.mode==='actual' && data[s.key]!==undefined && data[s.key]!==null){

        s.value = data[s.key];

      }

    });

    farm.actualStale = false;

    farm.lastActualOk = Date.now();

  }catch(e){

    // backend route missing or offline — keep last known actual values (passthrough)

    farm.actualStale = true;

  }

  if(farm.id===state.activeFarmId && state.activeTab==='dashboard') renderContent();

}



function activeOverrideFor(farm, sensorKey){

  if(!farm.activeScenario) return null;

  const cfg = farm.scenarios[farm.activeScenario];

  return cfg && cfg.overrides[sensorKey] ? cfg.overrides[sensorKey] : null;

}



function snapshotFarm(farm){

  const snap = {};

  farm.sensors.forEach(s=>{ snap[s.key]=s.value; });

  return snap;

}



function tickVirtual(farm){

  farm.sensors.forEach(s=>{

    const override = activeOverrideFor(farm, s.key);

    if(override){

      s.value = randIn(override.min, override.max, s.decimals);

    } else if(s.mode==='virtual'){

      s.value = randIn(s.min, s.max, s.decimals);

    }

    // actual-mode sensors with no active override keep whatever tickActual gave them

  });

  publishVirtual(farm);

  farm.history.push({ts:Date.now(), scenario:farm.activeScenario, snapshot:snapshotFarm(farm)});

  if(farm.history.length>200) farm.history.shift();

  if(farm.id===state.activeFarmId && state.activeTab==='dashboard') renderContent();

}



async function publishVirtual(farm){

  const payload = {};

  farm.sensors.forEach(s=>{

    if(s.mode==='virtual' || activeOverrideFor(farm,s.key)) payload[s.key]=s.value;

  });

  try{

    await fetch(virtualUrl(farm), {

      method:'POST',

      headers:{'Content-Type':'application/json'},

      body: JSON.stringify(payload)

    });

  }catch(e){ /* route not built yet — value still tracked locally for the demo */ }

}



function startFarmTimers(farm){

  stopFarmTimers(farm);

  farm.timers.actual = setInterval(()=>tickActual(farm), 3000);

  farm.timers.virtual = setInterval(()=>tickVirtual(farm), 4000);

  tickActual(farm); tickVirtual(farm);

}

function stopFarmTimers(farm){

  if(farm.timers.actual) clearInterval(farm.timers.actual);

  if(farm.timers.virtual) clearInterval(farm.timers.virtual);

}

function duplicateFarm(id){

  const src = state.farms.find(f=>f.id===id);

  if(!src) return;

  const farm = state.farms.find(f=>f.id===id);

  if(!farm) return;

  stopFarmTimers(farm);

  state.farms = state.farms.filter(f=>f.id!==id);

  if(state.activeFarmId===id){

    state.activeFarmId = state.farms[0] ? state.farms[0].id : null;

  }

  renderAll();

}

function handleImageUpload(evt){

  const farm = activeFarm(); if(!farm) return;

  const file = evt.target.files[0];

  if(!file) return;

  const reader = new FileReader();

  reader.onload = ()=>{

    farm.image = {dataUrl: reader.result, ts: Date.now()};

    renderContent();

  };

  reader.readAsDataURL(file);

  evt.target.value = '';

}

function addSensorFromForm(){

  const farm = activeFarm(); if(!farm) return;

  const name = document.getElementById('newSensorName').value.trim();

  const unit = document.getElementById('newSensorUnit').value.trim();

  const mode = document.getElementById('newSensorMode').value;

  const min = parseFloat(document.getElementById('newSensorMin').value);

  const max = parseFloat(document.getElementById('newSensorMax').value);

  if(!name || isNaN(min) || isNaN(max) || min>=max){

    alert('Give the sensor a name and a valid min < max range.');

    return;

  }

  const key = name.toLowerCase().replace(/[^a-z0-9]+/g,'_').replace(/^_|_$/g,'') || uid('sensor');

  if(farm.sensors.some(s=>s.key===key)){ alert('A sensor with this name already exists.'); return; }

  farm.sensors.push({key, name, unit, mode, min, max, decimals: (max-min)<5 ? 2 : 1, value: round((min+max)/2,1)});

  renderContent();

}



function removeSensor(key){

  const farm = activeFarm(); if(!farm) return;

  farm.sensors = farm.sensors.filter(s=>s.key!==key);

  Object.keys(farm.scenarios).forEach(s=>{ delete farm.scenarios[s].overrides[key]; });

  renderContent();

}



function updateSensorRange(key, field, value){

  const farm = activeFarm(); if(!farm) return;

  const sensor = farm.sensors.find(s=>s.key===key); if(!sensor) return;

  const v = parseFloat(value);

  if(!isNaN(v)) sensor[field]=v;

}



function toggleSensorMode(key){

  const farm = activeFarm(); if(!farm) return;

  const sensor = farm.sensors.find(s=>s.key===key); if(!sensor) return;

  sensor.mode = sensor.mode==='actual' ? 'virtual' : 'actual';

  renderContent();

}

function updateOverride(scenario, key, field, value){

  const farm = activeFarm(); if(!farm) return;

  const cfg = farm.scenarios[scenario];

  if(!cfg.overrides[key]) cfg.overrides[key] = {min:0,max:0};

  const v = parseFloat(value);

  if(!isNaN(v)) cfg.overrides[key][field]=v;

}



function toggleOverrideActive(scenario, key, checked){

  const farm = activeFarm(); if(!farm) return;

  const cfg = farm.scenarios[scenario];

  if(checked){

    if(!cfg.overrides[key]){

      const sensor = farm.sensors.find(s=>s.key===key);

      cfg.overrides[key] = {min:sensor.min, max:sensor.max};

    }

  } else {

    delete cfg.overrides[key];

  }

  renderContent();

}

  const farm = activeFarm(); if(!farm) return;

function quickSwitchScenario(name){

  const farm = activeFarm(); if(!farm) return;

  if(name===null){ deactivateScenario(); return; }

  const cfg = farm.scenarios[name];

  if(!cfg || Object.keys(cfg.overrides).length===0){

    // jump to settings to configure it first

    state.activeTab = 'settings';

    state.openScenario = name;

    renderAll();

    return;

  }

  applyScenario(name);

}

  farm.scenarios[clean] = {overrides:{}, saved:false};

  state.openScenario = clean;

  renderContent();

  if(BASE_SCENARIOS.includes(name)){ alert('Built-in scenarios can be reconfigured but not removed.'); return; }

  if(farm.activeScenario===name) farm.activeScenario=null;

  delete farm.scenarios[name];

  if(state.openScenario===name) state.openScenario=null;

  renderAll();

}

function rangeBar(sensor, override){

  const min = override ? override.min : sensor.min;

  const max = override ? override.max : sensor.max;

  const pct = clamp(((sensor.value-min)/((max-min)||1))*100, 0, 100);

  const cls = override ? 'overridden' : sensor.mode;

  return `

    <div class="rangebar ${cls}">

      <div class="rangebar-track">

        <div class="rangebar-fill" style="width:${pct}%"></div>

        <div class="rangebar-marker" style="left:${pct}%"></div>

      </div>

      <div class="rangebar-labels"><span>${fmt(min,sensor.decimals)}</span><span>${fmt(max,sensor.decimals)} ${sensor.unit}</span></div>

    </div>

  `;

}

function renderScenarioConfigSection(farm){

  const allNames = [...BASE_SCENARIOS, ...Object.keys(farm.scenarios).filter(n=>!BASE_SCENARIOS.includes(n))];

  const cards = allNames.map(name=>{

    const cfg = farm.scenarios[name];

    const color = scenarioColor(name, farm);

    const overrideCount = Object.keys(cfg.overrides).length;

    const isLive = farm.activeScenario===name;

    const isOpen = state.openScenario===name;

    const isCustom = !BASE_SCENARIOS.includes(name);

    return `

      <div class="scenario-card ${isOpen?'open':''} ${isLive?'live':''}" style="border-top-color:${color}" onclick="openScenario('${name}')">

        <div class="scenario-card-title"><span class="card-icon">${scenarioIcon(name)}</span>${name}</div>

        <div class="scenario-card-status">${overrideCount ? overrideCount+' sensor(s) configured' : 'Not configured'}${isCustom?' · custom':''}</div>

        ${isLive ? '<span class="live-badge">Active now</span>' : ''}

      </div>

    `;

  }).join('');



  let editor = '';

  if(state.openScenario && farm.scenarios[state.openScenario]){

    const name = state.openScenario;

    const cfg = farm.scenarios[name];

    const isLive = farm.activeScenario===name;

    const isCustom = !BASE_SCENARIOS.includes(name);

    editor = `

      <div class="scenario-editor">

        <div class="scenario-editor-head">

          <div>

            <h3>${name} configuration</h3>

            <div class="helper-text">Tick a sensor to override it whenever ${name} is applied. Every time this scenario runs, those sensors generate values inside the range you set here — everything else keeps behaving normally.</div>

          </div>

          <div class="scenario-editor-actions">

            ${isCustom ? `<button class="danger-btn" onclick="removeScenario('${name}')">Delete scenario</button>` : ''}

            <button onclick="saveScenario('${name}')">Save configuration</button>

            ${isLive

              ? `<button class="danger-btn" onclick="deactivateScenario()">Stop scenario</button>`

              : `<button class="primary" onclick="applyScenario('${name}')">Apply scenario</button>`}

          </div>

        </div>

        ${farm.sensors.map(s=>{

          const ov = cfg.overrides[s.key];

          const checked = !!ov;

          const min = ov ? ov.min : '';

          const max = ov ? ov.max : '';

          return `

            <div class="override-row">

              <div>

                <input type="checkbox" style="width:auto;display:inline-block;margin-right:8px;" ${checked?'checked':''} onchange="toggleOverrideActive('${name}','${s.key}',this.checked)">

                <span class="sname">${s.name} <span style="color:var(--ink-soft)">(${s.unit})</span></span>

              </div>

              <input type="number" step="any" placeholder="min" value="${min}" ${checked?'':'disabled'} onchange="updateOverride('${name}','${s.key}','min',this.value)">

              <input type="number" step="any" placeholder="max" value="${max}" ${checked?'':'disabled'} onchange="updateOverride('${name}','${s.key}','max',this.value)">

            </div>

          `;

        }).join('') || '<div class="empty-state">Add sensors first to configure this scenario.</div>'}

      </div>

    `;

  }



  return `

    <div class="panel-head" style="background:transparent;border:none;padding:0 0 10px;">

      <span class="panel-title" style="font-size:13px;">Scenarios for ${farm.name}</span>

    </div>

    <div class="scenario-grid">${cards}<div class="scenario-card custom-card" onclick="addCustomScenario()">+ Add custom scenario</div></div>

    ${editor}

  `;

}

(async function init(){

  const farm1 = createFarm('Farm 1', true);

  state.farms.push(farm1);

  state.activeFarmId = farm1.id;

  startFarmTimers(farm1);



  try{

    const res = await fetch('/farms');

    if(res.ok){

      const data = await res.json();

      (data.farms || []).forEach(f=>{

        const farm = createFarm(f.name || f.farm_id, false, {

          id: f.farm_id,

          meta: {

            field: f.field, crop: f.crop,

            plantingDate: new Date().toISOString().slice(0,10),

            daysSincePlanting: 0, growthStage: 'Seedling'

          }

        });

        state.farms.push(farm);

        startFarmTimers(farm);

      });

    }

  }catch(e){ /* backend unreachable — Farm 1 still works standalone */ }



  document.querySelector('[data-tab="dashboard"].tab-btn')?.classList.add('active');

  document.querySelector('[data-tab="dashboard"].side-nav-btn')?.classList.add('active');

  renderAll();

})();
